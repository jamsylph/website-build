def gm11_forecast(x0, n):
    """
    GM(1,1) forecasting model
    x0: Original sequence
    n: Number of points to predict
    """
    x1 = np.cumsum(x0)
    z1 = 0.5 * (x1[:-1] + x1[1:])
    
    B = np.vstack([-z1, np.ones_like(z1)]).T
    Y = x0[1:].reshape((-1, 1))
    
    # Solve for a, b using least squares
    [[a], [b]] = np.linalg.inv(B.T @ B) @ B.T @ Y
    
    # Forecast future values
    x_pred = [(x0[0] - b/a) * np.exp(-a * (i)) + b/a 
              for i in range(len(x0) + n)]
    
    return x_pred
